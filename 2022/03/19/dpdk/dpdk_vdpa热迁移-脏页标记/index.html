static __rte_always_inline void
vhost_log_page(uint8_t *log_base, uint64_t page)
{
    vhost_set_bit(page % 8, &log_base[page / 8]);
}

log_base жҜҸbitд»ЈиЎЁдёҖдёӘpage, 0иЎЁзӨәжІЎжңүж”№еҸҳиҝҮпјҢ1иЎЁзӨәж”№еҸҳиҝҮ

дҫӢеӯҗ:

еҒҮи®ҫ page зҡ„еҖјдёә 9пјҢиҝҷж„Ҹе‘ізқҖжҲ‘д»¬йңҖиҰҒж Үи®°з¬¬ 9 дёӘйЎөйқўе·Ідҝ®ж”№гҖӮ
и®Ўз®— page / 8 еҫ—еҲ° 1пјҢиЎЁзӨәlog_base[1]дёӯеӯҳеӮЁз¬¬ 9 дёӘйЎөйқўзҡ„зҠ¶жҖҒгҖӮ
и®Ўз®— page % 8 еҫ—еҲ° 1пјҢиЎЁзӨәз¬¬ 9 дёӘйЎөйқўзҡ„зҠ¶жҖҒз”Ёlog_base[1]дёӯзҡ„bit 1еӯҳеӮЁгҖӮ
